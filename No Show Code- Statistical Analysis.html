def probStatus(dataset, group_by):
    df = pds.crosstab(index = dataset[group_by], columns = dataset['No-show']).reset_index()
    df['probShowUp'] = df[0] / (df[0] + df[1])
    return df[[group_by, 'probShowUp']]
